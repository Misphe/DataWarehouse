WITH 
MEMBER [Measures].[Aktywne Stacje1] AS
    COUNT(
        FILTER(
            [ID Stacja Startowa].[ID Stacji].[ID Stacji].MEMBERS,
            NOT IsEmpty([Measures].[Wypozyczenie Count])
        )
    )

MEMBER [Measures].[Aktywne Stacje2] AS
    COUNT(
        FILTER(
            [ID Stacja Koncowa].[ID Stacji].[ID Stacji].MEMBERS,
            NOT IsEmpty([Measures].[Wypozyczenie Count])
        )
    )

MEMBER [Measures].[Suma Aktywnych Stacji] AS
    [Measures].[Aktywne Stacje1] + [Measures].[Aktywne Stacje2]

SELECT 
{
    [Measures].[Aktywne Stacje1]
} ON COLUMNS,
FILTER(
    [Data].[Nazwa Miesiaca].MEMBERS,
    [Data].[Nazwa Miesiaca].CURRENTMEMBER.NAME = "July" OR
	[Data].[Nazwa Miesiaca].CURRENTMEMBER.NAME = "June"
) ON ROWS
FROM [Przejazd]
where(
	[Data].[Rok].&[2023]
)
